const SHEET_NAME = "Memory Wall";
const ROOT_FOLDER_NAME = "HOPE21";

// Columns:
// A Timestamp
// B Name
// C Type
// D File Name
// E Drive Link / Main URL
// F Message
// G Status
// H File ID
// I Cloudinary URL
// J Cloudinary Public ID

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents || "{}");
    const sheet = getSheet();

    if (body.action === "uploadMemory") {
      const folders = getFolders();
      const name = body.name || "";
      const type = body.type || "";
      const fileName = body.fileName || "";
      const mimeType = body.mimeType || "";
      const base64 = body.base64 || "";

      const bytes = Utilities.base64Decode(base64);
      const blob = Utilities.newBlob(bytes, mimeType, fileName);

      const targetFolder = type === "video" ? folders.videos : folders.photos;
      const file = targetFolder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

      const status = type === "video" ? "PROCESSING" : "READY";

      sheet.appendRow([
        new Date(),
        name,
        type,
        file.getName(),
        file.getUrl(),
        "",
        status,
        file.getId(),
        "",
        ""
      ]);

      return json({ ok: true, url: file.getUrl(), status, fileId: file.getId() });
    }

    if (body.action === "saveCloudinaryVideo") {
      sheet.appendRow([
        new Date(),
        body.name || "",
        "video",
        body.fileName || "",
        body.cloudinaryUrl || "",
        "",
        "READY",
        "",
        body.cloudinaryUrl || "",
        body.publicId || ""
      ]);

      return json({ ok: true });
    }

    if (body.action === "saveMessage") {
      sheet.appendRow([
        new Date(),
        body.name || "",
        "message",
        "",
        "",
        body.message || "",
        "READY",
        "",
        "",
        ""
      ]);

      return json({ ok: true });
    }

    if (body.action === "listMemories") {
      return json({ ok: true, memories: getMemories(sheet) });
    }

    return json({ ok: false, error: "Unknown action" });

  } catch (err) {
    return json({ ok: false, error: String(err) });
  }
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);

  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow([
      "Timestamp",
      "Name",
      "Type",
      "File Name",
      "Drive Link",
      "Message",
      "Status",
      "File ID",
      "Cloudinary URL",
      "Cloudinary Public ID"
    ]);
    return sheet;
  }

  ensureHeaders(sheet);
  return sheet;
}

function ensureHeaders(sheet) {
  if (sheet.getLastColumn() < 10) {
    sheet.insertColumnsAfter(sheet.getLastColumn(), 10 - sheet.getLastColumn());
  }

  const required = [
    "Timestamp",
    "Name",
    "Type",
    "File Name",
    "Drive Link",
    "Message",
    "Status",
    "File ID",
    "Cloudinary URL",
    "Cloudinary Public ID"
  ];

  for (let i = 0; i < required.length; i++) {
    sheet.getRange(1, i + 1).setValue(required[i]);
  }

  const lastRow = sheet.getLastRow();
  if (lastRow <= 1) return;

  const range = sheet.getRange(2, 1, lastRow - 1, 10);
  const data = range.getValues();

  let changed = false;

  data.forEach(row => {
    const type = row[2];
    const driveLink = row[4];

    if (!row[6]) {
      row[6] = type === "video" ? "PROCESSING" : "READY";
      changed = true;
    }

    if (!row[7] && driveLink && !String(driveLink).includes("cloudinary.com")) {
      row[7] = extractDriveId(driveLink);
      changed = true;
    }

    if (!row[8] && type === "video" && String(driveLink).includes("cloudinary.com")) {
      row[8] = driveLink;
      changed = true;
    }
  });

  if (changed) range.setValues(data);
}

function getFolders() {
  const rootIterator = DriveApp.getFoldersByName(ROOT_FOLDER_NAME);
  const root = rootIterator.hasNext()
    ? rootIterator.next()
    : DriveApp.createFolder(ROOT_FOLDER_NAME);

  return {
    root,
    photos: getOrCreateFolder(root, "Photos"),
    videos: getOrCreateFolder(root, "Videos"),
    messages: getOrCreateFolder(root, "Messages")
  };
}

function getOrCreateFolder(parent, name) {
  const folders = parent.getFoldersByName(name);
  return folders.hasNext() ? folders.next() : parent.createFolder(name);
}

function getMemories(sheet) {
  ensureHeaders(sheet);

  const data = sheet.getDataRange().getValues();

  return data.slice(1).filter(row => row[0]).map(row => ({
    timestamp: row[0],
    name: row[1],
    type: row[2],
    fileName: row[3],
    driveLink: row[4],
    message: row[5],
    status: row[6] || (row[2] === "video" ? "PROCESSING" : "READY"),
    fileId: row[7] || extractDriveId(row[4]),
    cloudinaryUrl: row[8],
    cloudinaryPublicId: row[9]
  }));
}

function extractDriveId(url) {
  const match = String(url || "").match(/[-\w]{25,}/);
  return match ? match[0] : "";
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
